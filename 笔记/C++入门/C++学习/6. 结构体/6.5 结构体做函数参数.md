```cpp
// 结构体做函数参数

/*
 * 结构体做函数参数：将结构体作为参数传递给函数
 *
 * 传递方式：
 * 1. 值传递
 * 2. 地址传递
 *
 */

#include <iostream>
#include <string>
using namespace std;

// 定义学生结构体
struct student
{
    string name; // 姓名
    int age;     // 年龄
    int score;   // 成绩
};

// 打印学生信息函数

// 1.值传递
void printStudent1(student s)
{
    // 值传递，形参是实参的拷贝，对形参的修改不会影响实参
    cout << "值传递打印学生信息1" << endl;
    cout << "姓名：" << s.name << endl;
    cout << "年龄：" << s.age << endl;
    cout << "成绩：" << s.score << endl;
    cout << endl;

    s.age = 100; // 通过值传递修改形参的年龄，不会影响实参

    cout << "值传递打印学生信息2" << endl;
    cout << "姓名：" << s.name << endl;
    cout << "年龄：" << s.age << endl;
    cout << "成绩：" << s.score << endl;
    cout << endl;
}

// 2.地址传递
void printStudent2(student *s)
{
    // 地址传递，形参是实参的地址，对形参的修改会影响实参
    cout << "地址传递打印学生信息1" << endl;
    cout << "姓名：" << s->name << endl;
    cout << "年龄：" << s->age << endl;
    cout << "成绩：" << s->score << endl;
    cout << endl;

    s->age = 200; // 通过地址传递修改实参的年龄

    cout << "地址传递打印学生信息2" << endl;
    cout << "姓名：" << s->name << endl;
    cout << "年龄：" << s->age << endl;
    cout << "成绩：" << s->score << endl;
    cout << endl;
}

int main()
{
    // 结构体做函数参数
    // 将学生传入到一个参数中，打印学生身上的所有信息

    // 创建结构体变量
    struct student s;

    s.name = "小王";
    s.age = 20;
    s.score = 85;

    // 调用函数 >>> 1.值传递
    printStudent1(s);
    // 调用函数 >>> 2.地址传递
    printStudent2(&s);

    cout << "main函数中打印学生信息" << endl;
    cout << "姓名：" << s.name << endl;
    cout << "年龄：" << s.age << endl;
    cout << "成绩：" << s.score << endl;
}
```
