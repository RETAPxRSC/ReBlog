// 结构体中的const使用

// 作用：用const来防止误操作。

#include <iostream>
#include <string>
using namespace std;

struct Student
{
    string name; // 姓名
    int age;     // 年龄
    int score;   // 成绩
};

void printStudent(const Student *stu1)
{
    // const Student *stu1 ：stu1是一个指向const Student类型的指针，
    // 它指向的对象不能被修改，只能读取。
    cout << "姓名：" << stu1->name << endl;
    cout << "年龄：" << stu1->age << endl;
    cout << "成绩：" << stu1->score << endl;
    cout << endl;

    // stu1->age = 100; // 操作失败-不能通过指针修改const对象的成员

    cout << "姓名：" << stu1->name << endl;
    cout << "年龄：" << stu1->age << endl;
    cout << "成绩：" << stu1->score << endl;
    cout << endl;
}

int main()
{
    Student stu1;
    stu1.name = "张三";
    stu1.age = 18;
    stu1.score = 90;
    printStudent(&stu1);
    return 0;
}
