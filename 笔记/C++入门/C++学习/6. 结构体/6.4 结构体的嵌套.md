```cpp
#include <iostream>
#include <string>
using namespace std;

struct Student
{
string name; // 姓名
int age; // 年龄
int score; // 成绩
};

struct Teacher
{
int id;// 教师编号
string name; // 姓名
int age; // 年龄
struct Student Stu1; // 嵌套子结构体
};

int main()
{
// 定义老师
Teacher T;
T.id = 10000;
T.name = "老王";
T.age = 50;
// 定义嵌套的学生
T.Stu1.name = "小王";
T.Stu1.age = 20;
T.Stu1.score = 60;
// 输出老师和学生的信息
cout << "老师姓名：" << T.name << endl << "老师编号：" << T.id << endl;
cout << "学生姓名：" << T.Stu1.name << endl << "学生年龄：" << T.Stu1.age << endl << "学生成绩：" << T.Stu1.score << endl;
return 0;
}
```
